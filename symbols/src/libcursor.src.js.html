 76 /**
 77  * Same as moveForward or moveBackward depending on the sign of offset.
 78  * <pre>
 79  * move (9) is same as moveForward (9).
 80  * move (-9) is same as moveBackward (9).
 81  * </pre>
 82  * @see Cursor#moveForward
 83  * @see Cursor#moveBackward
 84  * @method
 85  * 
 86  * @param {integer} offset The amount by which cursor should be moved.
 87  */
 88 Cursor.prototype.move = function (offset) {
 89     if (offset > 0) return this.moveForward (offset);
 90     else return this.moveBackward (-offset);
 91 };
