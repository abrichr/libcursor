1 /**
  2  * Represents a Cursor.
  3  * @constructor
  4  * 
  5  * @param [DomElement]
  6  * This must be either an element with contenteditable set to true,
  7  * or a textarea or an input field.
  8  * If unspecified, this would return a cursor at the current visible
  9  * position of the cursor on the page.
 10  *
 11  * @param [position]
 12  * This specifies the position of the cursor in the DomElement.
 13  * When unspecified, while the DomElement is specified, the resultant
 14  * cursor will be at the start of the editing area. In case of unspecified
 15  * DomElement, the cursor at the current visible cursor position will be
 16  * returned.
 17  */
 18 function Cursor (DomElement, position) {
 19     // The case when nothing is specified, just get the current position
 20     // where cursor is in the document.
 21     if (arguments.length == 0) {
 22         // As of now, let us just handle text area.
 23         if (document.activeElement.nodeName == "TEXTAREA") {
 24             this.textarea = document.activeElement;
 25         }
 26     }
 27 }
 28 
 29 /**
 30  * Same as insertBefore
 31  * @see Cursor#insertBefore
 32  * @method
 33  */
 34 Cursor.prototype.insert = function (text) {
 35     return this.insertBefore (text);
 36 };

 39 /**
 40  * Inserts text at the position of the cursor, moves the cursor at the
 41  * end of the inserted text.
 42  * <pre>
 43  * Examples:
 44  * 
 45  *     +                                               +
 46  * big | apple -> insertBefore ('rotten') -> big rotten| apple
 47  *     +                                               +
 48  * 
 49  *      +-----+                                       +
 50  * this |isn't| good -> insertBefore ('is') -> this is| good
 51  *      +-----+                                       +
 52  * </pre>
 53  * @method
 54  * 
 55  * @param {string} text The text to be inserted.
 56  */
 57 Cursor.prototype.insertBefore = function (text) {
 58     this.textarea.setRangeText (text);
 59     this.textarea.setSelectionRange (
 60         this.textarea.selectionStart + text.length,
 61         this.textarea.selectionStart + text.length
 62     );
 63     return this;
 64 };
 65 
 66 /**
 67  * Inserts text at the position of the cursor, keeps the cursor at the
 68  * start of the inserted text. 
 69  * <pre>
 70  * Examples:
 71  * 
 72  *     +                                        +
 73  * big | apple -> insertAfter ('rotten') -> big |rotten apple
 74  *     +                                        +
 75  * 
 76  *      +-----+                                    +
 77  * this |isn't| good -> insertAfter ('is') -> this |is good
 78  *      +-----+                                    +
 79  * </pre>
 80  * @method
 81  * 
 82  * @param {string} text The text to be inserted.
 83  */
 84 Cursor.prototype.insertAfter = function (text) {
 85     this.textarea.setRangeText (text);
 86     this.textarea.setSelectionRange (this.textarea.selectionStart,
 87                                      this.textarea.selectionStart);
 88     return this;
 89 };
 90 
 91 /**
 92  * Same as moveForward or moveBackward depending on the sign of offset.
 93  * <pre>
 94  * move (9) is same as moveForward (9).
 95  * move (-9) is same as moveBackward (9).
 96  * </pre>
 97  * @see Cursor#moveForward
 98  * @see Cursor#moveBackward
 99  * @method
100  * 
101  * @param {integer} offset The amount by which cursor should be moved.
102  */
103 Cursor.prototype.move = function (offset) {
104     if (offset > 0) return this.moveForward (offset);
105     else return this.moveBackward (-offset);
106 };
107 
108 /**
109  * Moves the cursor forward by the amount specified.
110  * <pre>
111  * Examples:
112  * 
113  *         +                                         +
114  * this is |awesome -> moveForward (3) -> this is awe|some
115  *         +                                         +
116  *
117  *         +---+                                         +
118  * this is |awe|some -> moveForward (3) -> this is awesom|e
119  *         +---+                                         +
120  * 
121  * </pre>
122  * @method
123  * 
124  * @param {positive integer} offset The amount by which the cursor should
125  * move forward.
126  */
127 Cursor.prototype.moveForward = function (offset) {
128     this.textarea.setSelectionRange (this.textarea.selectionEnd + offset,
129                                      this.textarea.selectionEnd + offset);
130     return this;
131 };
132 
133 /**
134  * Moves the cursor backward by the amount specified.
135  * <pre>
136  * Examples:
137  * 
138  *         +                                    +
139  * this is |awesome -> moveBackward (3) -> this |is awesome
140  *         +                                    +
141  *
142  *         +---+                                 +
143  * this is |awe|some -> moveBackward (3) -> this |is awesome
144  *         +---+                                 +
145  * 
146  * </pre>
147  * @method
148  * 
149  * @param {positive integer} offset The amount by which the cursor should
150  * move backward.
151  */
152 Cursor.prototype.moveBackward = function (offset) {
153     this.textarea.setSelectionRange (this.textarea.selectionStart - offset,
154                                      this.textarea.selectionStart - offset);
155     return this;
156 };
157 
158 /**
159  * Same as deleteForward or deleteBackward depending on the sign of amount.
160  * <pre>
161  * delete (9) is same as deleteForward (9).
162  * delete (-9) is same as deleteBackward (9).
163  * </pre>
164  * @see Cursor#deleteForward
165  * @see Cursor#deleteBackward
166  * @method
167  * 
168  * @param {integer} amount The number of characters to be deleted.
169  */
170 Cursor.prototype.delete = function (amount) {
171     if (amount > 0) return this.deleteForward (amount);
172     else return this.deleteBackward (-amount);
173 };
174 
175 /**
176  * Deletes the specified number of characters ahead of the cursor.
177  * <pre>
178  * Examples:
179  * 
180  *         +                                        +
181  * this is |awesome -> deleteForward (3) -> this is |some
182  *         +                                        +
183  *
184  *         +---+                                        +
185  * this is |awe|some -> deleteForward (4) -> this is awe|
186  *         +---+                                        +
187  * 
188  * </pre>
189  * @method
190  * 
191  * @param {positive integer} amount The number of characters to be deleted.
192  */
193 Cursor.prototype.deleteForward = function (amount) {
194     this.textarea.setSelectionRange (this.textarea.selectionEnd,
195                                      this.textarea.selectionEnd + amount);
196     this.textarea.setRangeText ("");
197     return this;
198 };
199 
200 /**
201  * Deletes the specified number of characters before the cursor.
202  * <pre>
203  * Examples:
204  * 
205  *         +                                      +
206  * this is |awesome -> deleteBackward (3) -> this |awesome
207  *         +                                      +
208  *
209  *         +---+                                  +
210  * this is |awe|some -> deleteBackward (4) -> this|awesome
211  *         +---+                                  +
212  * 
213  * </pre>
214  * @method
215  * 
216  * @param {positive integer} amount The number of characters to be deleted.
217  */
218 Cursor.prototype.deleteBackward = function (amount) {
219     this.textarea.setSelectionRange (this.textarea.selectionStart - amount,
220                                      this.textarea.selectionStart);
221     this.textarea.setRangeText ("");
222     return this;
223 };
