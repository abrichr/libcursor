 10 /**
 11  * Static method to initialize and return an appropriate kind of cursor.
 12  * <pre>
 13  * 1) No arguments : Current cursor under currently focused element will be returned.
 14  * 
 15  * 2) Only DomElement specified : Cursor at the end of the text will be returned.
 16  * +-----------------------+
 17  * |                +      |
 18  * | text content...|      |
 19  * |                +      |
 20  * +-----------------------+
 21  * 
 22  * 3) DomElement and position both specified:
 23  * Cursor.new (document.getElementById('mydiv'), 5) will return following:
 24  * #mydiv
 25  * +-----------------+
 26  * |     +           |
 27  * |12345|6789..     |
 28  * |     +           |
 29  * +-----------------+
 30  * </pre>
 31  * @function
 32  * 
 33  * @param [DomElement]
 34  * This must be either an element with contenteditable set to true,
 35  * or a textarea.
 36  *
 37  * @param [position]
 38  * This specifies the position of the cursor in the DomElement.
 39  * 
 40  * @return An appropriately initialized Cursor object.
 41  */
 42 Cursor.new = function (DomElement, position) {
 43     // The case when nothing is specified, just get the current position
 44     // where cursor is in the document.
 45     if (arguments.length == 0) {
 46         // As of now, let us just handle text area.
 47         if (document.activeElement.nodeName == "TEXTAREA") {
 48             return new TextAreaCursor ();
 49         }
 50         else if (document.activeElement.isContentEditable) {
 51             return new ContentEditableCursor ();
 52         }
 53         else {
 54             // TODO: Raise an error.
 55         }
 56     }
 57     
 58     else if (arguments.length == 1) {
 59         if (DomElement.nodeName == "TEXTAREA") {
 60             return new TextAreaCursor (DomElement);
 61         }
 62         else if (DomElement.isContentEditable) {
 63             return new ContentEditableCursor (DomElement);
 64         }
 65         else {
 66             // TODO: Raise an error.
 67         }
 68     }
 69     
 70     else if (arguments.length == 2) {
 71         if (DomElement.nodeName == "TEXTAREA") {
 72             return new TextAreaCursor (DomElement, position);
 73         }
 74         else if (document.activeElement.isContentEditable) {
 75             return new ContentEditableCursor (DomElement, position);
 76         }
 77         else {
 78             // TODO: Raise an error.
 79         }
 80     }
 81     
 82     return null;
 83     // TODO: Raise an error.
 84 };

 86 function TextAreaCursor (Textarea, position) {
 87     if (arguments.length == 0) {
 88         this.textarea = document.activeElement;
 89     }
 90     else if (arguments.length == 1) {
 91         this.textarea = Textarea;
 92         this.textarea.setSelectionRange (this.textarea.value.length,
 93                                          this.textarea.value.length);
 94     }
 95     else if (arguments.length == 2) {
 96         this.textarea = Textarea;
 97         this.textarea.setSelectionRange (position,
 98                                          position);
 99     }
100 }

102 function countTextNodes (elem) {
103     var count = 0;
104     for (var i = 0; i < elem.childNodes.length; i++) {
105         if (elem.childNodes[i].nodeType == Node.TEXT_NODE) count++;
106     }
107     return count;
108 }

110 function ContentEditableCursor (DomElement, position) {
111     if (arguments.length == 0) {
112         this.selection = document.getSelection ();
113     }
114     else if (arguments.length == 1) {
115         DomElement.focus();
116         this.selection = document.getSelection ();
117         this.selection.extend (
118             DomElement,
119             DomElement.children.length + countTextNodes (DomElement)
120         );
121         this.selection.collapseToEnd ();
122     }
123     else if (arguments.length == 2) {
124         DomElement.focus();
125         this.selection = document.getSelection ();
126         this.selection.extend (DomElement, 0);
127         this.selection.collapseToStart ();
128         for (var i = 0; i < position; i++) {
129             this.selection.modify ('move',
130                                    'forward',
131                                    'character');
132         }
133     }
134 }

168 TextAreaCursor.prototype.insertBefore = function (text) {
169     this.textarea.setRangeText (text);
170     this.textarea.setSelectionRange (
171         this.textarea.selectionStart + text.length,
172         this.textarea.selectionStart + text.length
173     );
174     return this;
175 };
176 
177 ContentEditableCursor.prototype.insertBefore = function (text) {
178     this.insertAfter (text);
179     this.move (text.length);
180     return this;
181 };

204 TextAreaCursor.prototype.insertAfter = function (text) {
205     this.textarea.setRangeText (text);
206     this.textarea.setSelectionRange (this.textarea.selectionStart,
207                                      this.textarea.selectionStart);
208     return this;
209 };
210 
211 ContentEditableCursor.prototype.insertAfter = function (text) {
212     var range = this.selection.getRangeAt (0);
213     range.deleteContents ();
214     range.insertNode (document.createTextNode (text));
215     this.selection.collapseToStart ();
216     return this;
217 };

400 /**
401  * Get the text surrounding the cursor.
402  * <pre>
403  * Examples:
404  * 
405  *         +      
406  * this is |awesome
407  *         +     
408  * -> getText (-3, 3) "is awe"
409  * -> getText (-4, 0) " is "
410  * -> getText (2, 4)  "es"
411  * 
412  *         +---+
413  * this is |awe|some
414  *         +---+
415  * -> getText (-3, 3) "is awesom"
416  * -> getText (-4, 0) " is awe"
417  * -> getText (2, 4)  "esome"
418  * 
419  * </pre>
420  * <b>Note:</b> The cursor itself does not move.
421  * @method
422  * 
423  * @param {itive integer} start Start of the cursor moves this much forward.
424  * @param {itive integer} end End of the cursor moves this much forward.
425  * @return {string} The text enclosed between the start and end of the cursor after the (imaginary) moving is done.
426  */
427 Cursor.prototype.getText = function (start, end) {};
428 
429 TextAreaCursor.prototype.getText = function (start, end) {
430     var new_start = this.textarea.selectionStart + start;
431     var new_end = this.textarea.selectionEnd + end;
432     return this.textarea.value.substring (new_start, new_end);
433 };
434 
435 ContentEditableCursor.prototype.getText = function (start, end) {
436     var range = this.selection.getRangeAt (0);
437     return this.selection.anchorNode.textContent.substring (
438         range.startOffset + start,
439         range.endOffset + end
440     );
441 };
